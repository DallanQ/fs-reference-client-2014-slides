MV* Architecture

tree.js


(function(){
  'use strict';
  angular.module('tree', ['ionic', 'rcFamily', 'rcFS', 'rcFSUtils', 'lodash'])
  .config(function ($stateProvider) {
    $stateProvider.state('tree', {
      url: '/tree',
      controller: 'TreeController as tree',
      templateUrl: 'js/tree/tree.html',
      resolve: {
        families: function(rcFS, rcFSUtils) {
          return rcFS.getCurrentUser().then(function(response) {
            var personId = response.getUser().personId;
            return rcFS.getSpouses(personId).then(function(response) {
              var spouseId = response.getCoupleRelationships()[0].$getSpouseId(personId);
              return rcFS.getAncestry(personId, {spouse: spouseId, generations: 5}).then(function(response) {
                return rcFSUtils.pedigreeToFamilies(response);
              });
            });
          });
        }
      }
    });
  })
  .controller('TreeController', function (_, families) {
    this.families = _.map(families, function(family) {
      return _.extend({
        numStars: Math.floor(Math.random() * 6),
        numChallenges: Math.floor(Math.random() * 3)
      }, family);
    });
  });
})();
